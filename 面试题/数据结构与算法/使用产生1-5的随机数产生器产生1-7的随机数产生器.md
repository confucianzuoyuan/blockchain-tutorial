func rand7() int {
    vals := [][]int{
       []int{1, 2, 3, 4, 5},
       []int{6, 7, 1, 2, 3},
       []int{4, 5, 6, 7, 1},
       []int{2, 3, 4, 5, 6},
       []int{7, 0, 0, 0, 0},
    }

    result := 0
    for result == 0 {
        i := rand5()
        j := rand5()
        result = vals[i-1][j-1]
    }
    return result
}
